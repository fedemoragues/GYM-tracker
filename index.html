<script>

const GOAL = 3, TARGET = 80, KEY = 'gym_standalone_v1';


const PRE = [

  '2025-12-30','2026-01-02','2026-01-03',

  '2026-01-07','2026-01-09','2026-01-10',

  '2026-01-14','2026-01-15','2026-01-17',

  '2026-01-21','2026-01-22','2026-01-24',

  '2026-01-28','2026-01-29','2026-01-31',

  '2026-02-04','2026-02-05','2026-02-07',

  '2026-02-11','2026-02-12','2026-02-14',

  '2026-02-18','2026-02-21',

  '2026-02-25','2026-02-26','2026-02-28',

  '2026-03-04','2026-03-05','2026-03-07',

  '2026-03-11','2026-03-14',

  '2026-03-18','2026-03-19','2026-03-21',

  '2026-03-25','2026-03-29',

  '2026-04-01','2026-04-02','2026-04-04',

];


function wk(d) {

  const dt = new Date(d + 'T12:00:00'), day = dt.getDay();

  const mon = new Date(dt);

  mon.setDate(dt.getDate() - (day === 0 ? 6 : day - 1));

  return mon.toISOString().slice(0,10);

}

function mon(w) { return new Date(w + 'T12:00:00'); }

function sun(w) { const m = mon(w), s = new Date(m); s.setDate(m.getDate()+6); return s; }

function fmt(d) { return d.toLocaleDateString('es-AR',{day:'numeric',month:'short'}); }

function today() {

  const n = new Date();

  return `${n.getFullYear()}-${String(n.getMonth()+1).padStart(2,'0')}-${String(n.getDate()).padStart(2,'0')}`;

}

function load() { try { const r = localStorage.getItem(KEY); if(r) return JSON.parse(r); } catch{} return null; }

function save(s) { try { localStorage.setItem(KEY, JSON.stringify(s)); } catch{} }


let sessions = load() ?? PRE;

save(sessions);


function showToast(msg) {

  const t = document.getElementById('toast');

  t.textContent = msg; t.classList.add('show');

  setTimeout(() => t.classList.remove('show'), 1800);

}


function render() {

  const td = today(), cw = wk(td);

  const allW = [...new Set(sessions.map(s => wk(s)))].sort();

  const past = allW.filter(w => w < cw);

  const passed = past.length;

  const completed = past.filter(w => sessions.filter(s => wk(s) === w).length >= GOAL).length;

  const pct = passed === 0 ? null : Math.round(completed / passed * 100);


  document.getElementById('m-weeks').textContent = passed;

  document.getElementById('m-ok').textContent = completed;

  const pEl = document.getElementById('m-pct');

  if (pct === null) { pEl.textContent = '—'; pEl.style.color = '#888'; }

  else { pEl.textContent = pct + '%'; pEl.style.color = pct >= TARGET ? '#4ade80' : pct >= 60 ? '#fbbf24' : '#f87171'; }


  const cur = sessions.filter(s => wk(s) === cw), cnt = cur.length;

  document.getElementById('week-title').textContent = 'Semana actual · ' + fmt(mon(cw)) + ' – ' + fmt(sun(cw));


  const pips = document.getElementById('pips');

  pips.innerHTML = '';

  for (let i = 0; i < Math.max(GOAL, cnt); i++) {

    const p = document.createElement('div');

    p.className = 'pip ' + (i < cnt ? 'pip-done' : 'pip-empty');

    p.textContent = i < cnt ? '✓' : '';

    pips.appendChild(p);

  }


  const st = document.getElementById('week-status');

  if (cnt >= GOAL) { st.textContent = '¡Semana cumplida! ' + cnt + '/' + GOAL; st.className = 'week-status status-done'; }

  else { const r = GOAL - cnt; st.textContent = 'Faltan ' + r + ' ida' + (r > 1 ? 's' : '') + ' para cumplir'; st.className = 'week-status status-partial'; }


  const dots = document.getElementById('dots');

  dots.innerHTML = '';

  past.slice(-32).forEach(w => {

    const c = sessions.filter(s => wk(s) === w).length;

    const d = document.createElement('div');

    d.className = 'dot ' + (c >= GOAL ? 'dot-done' : 'dot-fail');

    d.title = fmt(mon(w)) + ': ' + c;

    d.textContent = c;

    dots.appendChild(d);

  });


  const hist = document.getElementById('history');

  hist.innerHTML = '';

  [...sessions].sort((a,b) => b.localeCompare(a)).slice(0,40).forEach(s => {

    const d = new Date(s + 'T12:00:00'), el = document.createElement('div');

    el.className = 'history-item';

    el.textContent = d.toLocaleDateString('es-AR',{weekday:'short',day:'numeric',month:'short',year:'numeric'});

    hist.appendChild(el);

  });

}


function addSession() {

  sessions.push(today());

  save(sessions);

  render();

  showToast('💪 ¡Registrado!');

}


function undoLast() {

  if (!sessions.length) return;

  sessions.pop(); save(sessions); render();

  showToast('↩ Deshecho');

}


render();
